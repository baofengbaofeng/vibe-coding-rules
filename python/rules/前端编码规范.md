## 编码规范

### 文件命名
- **组件文件**: 大驼峰命名，如 `UserProfile.vue`
- **工具文件**: 小驼峰命名，如 `formatDate.ts`
- **类型文件**: 小驼峰命名 + `.d.ts`，如 `user.d.ts`
- **样式文件**: 小写字母 + 连字符，如 `user-profile.less`

### 组件规范

#### 单文件组件结构
```
<template>
  <!-- 模板内容 -->
  <div class="user-profile">
    <van-cell-group>
      <van-cell title="用户名" :value="user.name" />
      <van-cell title="邮箱" :value="user.email" />
    </van-cell-group>
  </div>
</template>

<script setup lang="ts">
// 必须使用 Composition API + <script setup>
import { ref, computed, onMounted } from 'vue'
import { useUserStore } from '@/stores/user'
import { Cell, CellGroup } from 'vant'
import type { User } from '@/types/user'

// Props定义（必须使用TypeScript）
interface Props {
  userId: number
  showAvatar?: boolean
}

const props = withDefaults(defineProps<Props>(), {
  showAvatar: true
})

// Emits定义
const emit = defineEmits<{
  (e: 'update:user', user: User): void
  (e: 'delete'): void
}>()

// 状态管理
const userStore = useUserStore()
const loading = ref(false)
const user = ref<User | null>(null)

// 计算属性
const fullName = computed(() => {
  if (!user.value) return ''
  return `${user.value.firstName} ${user.value.lastName}`
})

// 生命周期
onMounted(() => {
  loadUser()
})

// 方法
async function loadUser() {
  try {
    loading.value = true
    user.value = await userStore.fetchUser(props.userId)
  } catch (error) {
    console.error('Failed to load user:', error)
  } finally {
    loading.value = false
  }
}

function handleUpdate() {
  if (user.value) {
    emit('update:user', user.value)
  }
}

// 暴露给模板的方法
defineExpose({
  loadUser,
  handleUpdate
})
</script>

<style scoped lang="less">
// 必须使用scoped样式
.user-profile {
  padding: 16px;
  
  :deep(.van-cell) {  // 深度选择器
    background-color: #f5f5f5;
  }
}
</style>
```

1. **单一职责**: 每个组件只负责一个功能

3. **可测试性**: 组件应该易于测试
4. **类型安全**: 必须使用TypeScript类型注解

### 状态管理（Pinia）

#### Store定义
```
// stores/user.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { User, UserLoginParams } from '@/types/user'
import { userApi } from '@/api/users'

export const useUserStore = defineStore('user', () => {
  // State
  const user = ref<User | null>(null)
  const token = ref<string>('')
  const loading = ref(false)
  
  // Getters
  const isLoggedIn = computed(() => !!token.value)
  const userName = computed(() => user.value?.name || '')
  
  // Actions
  async function login(params: UserLoginParams) {
    try {
      loading.value = true
      const response = await userApi.login(params)
      
      // 更新状态
      token.value = response.token
      user.value = response.user
      
      // 持久化存储
      localStorage.setItem('token', response.token)
      
      return response
    } catch (error) {
      console.error('Login failed:', error)
      throw error
    } finally {
      loading.value = false
    }
  }
  
  async function logout() {
    token.value = ''
    user.value = null
    localStorage.removeItem('token')
  }
  
  async function fetchUser(userId: number) {
    const response = await userApi.getUser(userId)
    user.value = response.data
    return response.data
  }
  
  // 返回状态和方法
  return {
    // State
    user,
    token,
    loading,
    
    // Getters
    isLoggedIn,
    userName,
    
    // Actions
    login,
    logout,
    fetchUser
  }
})
```

#### Store使用规范
1. **模块化**: 按功能模块划分Store
2. **类型安全**: 必须定义类型
3. **异步处理**: 使用async/await处理异步
4. **错误处理**: 必须处理错误情况

### API层规范

#### 请求封装
```
// utils/request.ts
import axios from 'axios'
import type { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { showToast } from 'vant'

// 创建axios实例
const request: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    // 添加token
    const token = localStorage.getItem('token')
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
request.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, message, data } = response.data
    
    // 业务成功
    if (code === 0) {
      return data
    }
    
    // 业务错误
    showToast(message || '请求失败')
    return Promise.reject(new Error(message))
  },
  (error) => {
    // 网络错误
    if (error.response?.status === 401) {
      // 未授权，跳转到登录页
      window.location.href = '/login'
    } else if (error.response?.status === 403) {
      showToast('权限不足')
    } else {
      showToast('网络错误，请稍后重试')
    }
    
    return Promise.reject(error)
  }
)

export default request
```

#### API模块
```
// api/users.ts
import request from '@/utils/request'
import type { User, UserLoginParams, ApiResponse } from '@/types/api'

export const userApi = {
  // 登录
  login(params: UserLoginParams): Promise<ApiResponse<{ token: string; user: User }>> {
    return request.post('/auth/login', params)
  },
  
  // 获取用户信息
  getUser(userId: number): Promise<ApiResponse<User>> {
    return request.get(`/users/${userId}`)
  },
  
  // 更新用户信息
  updateUser(userId: number, data: Partial<User>): Promise<ApiResponse<User>> {
    return request.put(`/users/${userId}`, data)
  },
  
  // 删除用户
  deleteUser(userId: number): Promise<ApiResponse<void>> {
    return request.delete(`/users/${userId}`)
  }
}
```
