### 3. Send + Sync约束
```
use std::sync::Arc;
use tokio::sync::Mutex;

/// 需要在多线程间共享的类型必须实现Send + Sync
pub struct SharedService {
    // Arc<T>要求T: Send + Sync
    cache: Arc<Mutex<HashMap<String, String>>>,
    // RwLock<T>要求T: Send + Sync
    config: Arc<tokio::sync::RwLock<Config>>,
}

impl SharedService {
    pub fn new() -> Self {
        Self {
            cache: Arc::new(Mutex::new(HashMap::new())),
            config: Arc::new(tokio::sync::RwLock::new(Config::default())),
        }
    }
    
    /// 异步方法需要Send约束
    pub async fn get_data(&self, key: &str) -> Result<Option<String>, Error> {
        let cache = self.cache.lock().await;
        Ok(cache.get(key).cloned())
    }
}

/// 确保Future是Send的
pub async fn spawn_task() -> Result<(), Error> {
    let service = Arc::new(SharedService::new());
    
    // spawn要求Future: Send + 'static
    tokio::spawn(async move {
        let result = service.get_data("key").await;
        // 处理结果
    });
    
    Ok(())
}

## 并发模式

### 1. 并行任务
```
use tokio::task;
use futures::future::join_all;

/// 并行执行多个异步任务
pub async fn process_users_parallel(users: Vec<User>) -> Result<Vec<ProcessedUser>, Error> {
    let mut tasks = Vec::new();
    
    for user in users {
        // 为每个用户创建独立任务
        let task = task::spawn(async move {
            process_user(user).await
        });
        tasks.push(task);
    }
    
    // 等待所有任务完成
    let mut results = Vec::new();
    for task in tasks {
        match task.await {
            Ok(Ok(processed)) => results.push(processed),
            Ok(Err(e)) => return Err(e),
            Err(e) => return Err(Error::from(e)),
        }
    }
    
    Ok(results)
}

/// 使用join_all简化并行处理
pub async fn process_batch_parallel(items: Vec<Item>) -> Result<Vec<Result>, Error> {
    let futures = items.into_iter().map(|item| async move {
        process_item(item).await
    });
    
    let results = join_all(futures).await;
    
    // 检查是否有错误
    for result in &results {
        if let Err(e) = result {
            // 记录错误但继续处理其他项
            tracing::error!("处理失败: {}", e);
        }
    }
    
    Ok(results.into_iter().filter_map(Result::ok).collect())
}
```

use futures::stream::{self, StreamExt};

pub async fn process_stream(users: Vec<User>) -> Result<Vec<ProcessedUser>, Error> {
    stream::iter(users)
        .map(|user| async move {
            process_user(user).await
        })
        .buffered(5)  // 并发度为5
        .filter_map(|result| async move {
            match result {
                Ok(processed) => Some(processed),
                Err(e) => {
                    tracing::warn!("处理失败: {}", e);
                    None
                }
            }
        })
        .collect()
        .await
}

/// 带优先级的select!
pub async fn fetch_with_fallback() -> Result<Data, Error> {
    select! {
        // 优先尝试缓存
        result = fetch_from_cache() => {
            match result {
                Ok(data) => Ok(data),
                Err(_) => {
                    // 缓存失败，回退到数据库
                    fetch_from_database().await
                }
            }
        }
        // 超时保护
        _ = sleep(Duration::from_secs(2)) => {
            Err(Error::Timeout)
        }
    }
}

/// 循环中的select!
pub async fn process_with_cancellation(mut rx: tokio::sync::mpsc::Receiver<Message>) {
    let mut shutdown = false;
    
    while !shutdown {
        select! {
            // 处理消息
            Some(message) = rx.recv() => {
                process_message(message).await;
            }
            // 检查关闭信号
            _ = tokio::signal::ctrl_c() => {
                tracing::info!("收到关闭信号");
                shutdown = true;
            }
            // 超时检查
            _ = sleep(Duration::from_secs(60)) => {
                tracing::debug!("心跳检查");
            }
        }
    }
    
    tracing::info!("优雅关闭完成");
}

pub enum AppError {

pub async fn complex_operation() -> Result<Data, AppError> {
    // 异步错误传播
    let user = fetch_user(1).await?;
    let profile = fetch_profile(user.id).await?;
    let data = process_data(&profile).await?;
    
    Ok(data)
}

/// 错误转换
pub async fn fetch_with_retry(url: &str) -> Result<String, AppError> {
    for attempt in 1..=3 {
        match reqwest::get(url).await {
            Ok(response) => {
                return response.text()
                    .await
                    .map_err(AppError::Network);
            }
            Err(e) if attempt == 3 => {
                return Err(AppError::Network(e));
            }
            Err(_) => {
                tokio::time::sleep(Duration::from_secs(attempt as u64)).await;
                continue;
            }
        }
    }
    
    unreachable!()
}

use tokio::time::{timeout, Duration};

/// 带超时的异步操作
pub async fn fetch_with_deadline(url: &str) -> Result<String, Error> {
    let timeout_duration = Duration::from_secs(5);
    
    match timeout(timeout_duration, reqwest::get(url)).await {
        Ok(Ok(response)) => {

        Ok(Err(e)) => Err(Error::Network(e)),
        Err(_) => Err(Error::Timeout),

/// 多个操作的超时控制
pub async fn complex_operation_with_timeout() -> Result<Data, Error> {
    let overall_timeout = Duration::from_secs(10);
    
    timeout(overall_timeout, async {
        let step1 = step_one().await?;
        let step2 = step_two(&step1).await?;
        step_three(&step2).await
    })
    .await
    .map_err(|_| Error::Timeout)?
}

pub struct TimeoutConfig {

    pub read_timeout: Duration,
    pub write_timeout: Duration,

        F: Future<Output = Result<T, Error>>,

use sqlx::postgres::{PgPool, PgPoolOptions};
use std::sync::Arc;

/// 数据库连接池管理
pub struct Database {
    pool: Arc<PgPool>,
}

impl Database {
    pub async fn new(database_url: &str) -> Result<Self, Error> {
        let pool = PgPoolOptions::new()
            .max_connections(20)
            .min_connections(5)
            .max_lifetime(Duration::from_secs(30 * 60))
            .idle_timeout(Duration::from_secs(10 * 60))
            .connect(database_url)
            .await?;
        
        Ok(Self {
            pool: Arc::new(pool),
        })
    }
    
    /// 获取连接（自动管理生命周期）
    pub async fn get_connection(&self) -> Result<sqlx::pool::PoolConnection<sqlx::Postgres>, Error> {
        self.pool.acquire().await.map_err(Into::into)
    }
    
    /// 在事务中执行操作
    pub async fn in_transaction<F, T>(&self, f: F) -> Result<T, Error>
    where
        F: for<'c> FnOnce(&mut sqlx::Transaction<'c, sqlx::Postgres>) -> Pin<Box<dyn Future<Output = Result<T, Error>> + Send + 'c>>,
    {
        let mut tx = self.pool.begin().await?;
        
        let result = f(&mut tx).await;
        
        match result {
            Ok(value) => {
                tx.commit().await?;
                Ok(value)
            }
            Err(e) => {
                tx.rollback().await?;
                Err(e)
            }
        }
    }
}
```

/// API限流器
pub struct RateLimiter {

}

impl RateLimiter {
    pub fn new(requests_per_minute: u32) -> Self {
        let quota = Quota::per_minute(NonZeroU32::new(requests_per_minute).unwrap());

    pub fn try_check(&self) -> Result<(), Error> {
        self.limiter.check()
            .map(|_| ())
            .map_err(|_| Error::RateLimited)
    }
}

use bytes::Bytes;

use tokio::io::{AsyncReadExt, AsyncWriteExt, BufWriter};

/// 流式处理大文件
pub async fn process_large_file_streaming(
    input_path: &str,
    output_path: &str,
) -> Result<(), Error> {
    let mut input = File::open(input_path).await?;
    let output = File::create(output_path).await?;
    let mut writer = BufWriter::new(output);
    
    let mut buffer = vec![0u8; 8192];  // 8KB缓冲区
    
    loop {
        let bytes_read = input.read(&mut buffer).await?;
        if bytes_read == 0 {
            break;
        }
        
        // 处理数据块
        let processed = process_chunk(&buffer[..bytes_read]).await?;
        
        // 写入处理后的数据
        writer.write_all(&processed).await?;
    }
    
    writer.flush().await?;
    Ok(())
}

/// 使用Bytes避免拷贝
pub async fn process_bytes_efficiently(data: Bytes) -> Result<Bytes, Error> {
    // Bytes支持零拷贝切片
    let header = data.slice(0..4);
    let body = data.slice(4..);
    
    // 处理数据
    let processed_header = process_header(&header).await?;
    let processed_body = process_body(&body).await?;
    
    // 重新组合
    let mut result = BytesMut::new();
    result.extend_from_slice(&processed_header);
    result.extend_from_slice(&processed_body);
    
    Ok(result.freeze())
}

/// 批量数据库操作
pub async fn batch_insert_users(users: Vec<User>) -> Result<(), Error> {
    if users.is_empty() {
        return Ok(());
    }
    
    // 分批处理，每批1000条
    for chunk in users.chunks(1000) {
        let mut query_builder: QueryBuilder<Postgres> = QueryBuilder::new(
            "INSERT INTO users (username, email) "
        );
        
        query_builder.push_values(chunk, |mut b, user| {
            b.push_bind(&user.username);
            b.push_bind(&user.email);
        });
        
        let query = query_builder.build();
        query.execute(&self.pool).await?;
    }
    
    Ok(())
}

/// 批量API调用
pub async fn batch_api_requests(urls: Vec<String>) -> Result<Vec<String>, Error> {
    let client = reqwest::Client::new();
    let mut tasks = Vec::new();
    
    // 创建所有请求的Future
    for url in urls {
        let client = client.clone();
        let task = tokio::spawn(async move {
            client.get(&url).send().await?.text().await
        });
        tasks.push(task);
    }
    
    // 并发执行所有请求
    let mut results = Vec::new();
    for task in tasks {
        match task.await {
            Ok(Ok(content)) => results.push(content),
            Ok(Err(e)) => return Err(Error::Network(e)),
            Err(e) => return Err(Error::from(e)),
        }
    }
    
    Ok(results)
}

use moka::future::Cache;
use std::time::Duration;

/// 异步缓存
pub struct AsyncCache {
    cache: Cache<String, String>,
}

impl AsyncCache {
    pub fn new() -> Self {
        let cache = Cache::builder()
            .max_capacity(10_000)
            .time_to_live(Duration::from_secs(300))
            .time_to_idle(Duration::from_secs(60))
            .build();
        
        Self { cache }
    }
    
    /// 获取或计算缓存值
    pub async fn get_or_compute(&self, key: String) -> Result<String, Error> {
        self.cache
            .try_get_with(key, async {
                compute_value(&key).await
            })
            .await
            .map_err(|e| Error::Cache(e.to_string()))
    }
    
    /// 批量获取
    pub async fn get_many(&self, keys: Vec<String>) -> Result<HashMap<String, String>, Error> {
        let mut results = HashMap::new();
        let mut missing_keys = Vec::new();
        
        // 先检查缓存
        for key in keys {
            if let Some(value) = self.cache.get(&key).await {
                results.insert(key, value);
            } else {
                missing_keys.push(key);
            }
        }
        
        // 批量计算缺失的值
        if !missing_keys.is_empty() {
            let computed = compute_many(&missing_keys).await?;
            
            // 更新缓存
            for (key, value) in &computed {
                self.cache.insert(key.clone(), value.clone()).await;
                results.insert(key.clone(), value.clone());
            }
        }
        
        Ok(results)
    }
}

pub async fn process_complex_workflow() -> Result<Output, Error> {
    // 将长链拆分为多个有意义的步骤
    let input = validate_input().await?;
    let processed = process_stage_one(input).await?;
    let result = process_stage_two(processed).await?;
    
    Ok(result)
}

/// 合理使用.await
pub async fn efficient_await_usage() -> Result<(), Error> {
    // 可以并行执行的操作
    let future1 = operation_one();
    let future2 = operation_two();
    
    // 使用join并行等待
    let (result1, result2) = tokio::join!(future1, future2);
    result1?;
    result2?;
    
    // 顺序依赖的操作
    let step1 = step_one().await?;
    let step2 = step_two(&step1).await?;
    
    Ok(())
}

            Ok(data) => return Ok(data),

/// RAII模式管理资源
pub struct ManagedConnection {
    connection: Option<Connection>,
}

impl ManagedConnection {
    pub async fn new() -> Result<Self, Error> {
        let connection = establish_connection().await?;
        Ok(Self {
            connection: Some(connection),
        })
    }
    
    pub async fn execute(&mut self, query: &str) -> Result<(), Error> {
        if let Some(conn) = &mut self.connection {
            conn.execute(query).await
        } else {
            Err(Error::ConnectionClosed)
        }
    }
}

impl Drop for ManagedConnection {
    fn drop(&mut self) {
        if let Some(conn) = self.connection.take() {
            // 异步清理（在实际应用中可能需要特殊处理）
            tokio::spawn(async move {
                let _ = conn.close().await;
            });
        }
    }
}

## 常见问题与解决方案

### 1. 死锁问题
```
/// 避免异步死锁
pub async fn avoid_deadlock() -> Result<(), Error> {
    let lock1 = Arc::new(Mutex::new(()));
    let lock2 = Arc::new(Mutex::new(()));
    
    // 错误的顺序可能导致死锁
    // let _guard1 = lock1.lock().await;
    // let _guard2 = lock2.lock().await;
    
    // 正确的做法：按固定顺序获取锁
    let guard1 = lock1.lock().await;
    let guard2 = lock2.lock().await;
    
    // 或者使用try_lock
    match lock1.try_lock() {
        Ok(_guard1) => {
            match lock2.try_lock() {
                Ok(_guard2) => Ok(()),
                Err(_) => Err(Error::WouldBlock),
            }
        }
        Err(_) => Err(Error::WouldBlock),
    }
}
```

### 2. 内存泄漏
```
/// 避免异步内存泄漏
pub async fn avoid_memory_leak() -> Result<(), Error> {
    // 使用Weak引用打破循环引用
    let strong = Arc::new(Mutex::new(Data::new()));
    let weak = Arc::downgrade(&strong);
    
    tokio::spawn(async move {
        if let Some(strong_ref) = weak.upgrade() {
            // 使用强引用
            let _data = strong_ref.lock().await;
        }
        // weak引用不会阻止Arc被释放
    });
    
    Ok(())
}
```

use tokio::sync::watch;

/// 使用watch通道避免竞态条件
pub async fn avoid_race_condition() -> Result<(), Error> {
    let (tx, mut rx) = watch::channel(0);
    
    // 生产者任务
    let producer = tokio::spawn(async move {
        for i in 1..=10 {
            tx.send(i).unwrap();
            tokio::time::sleep(Duration::from_millis(100)).await;
        }
    });
    
    // 消费者任务
    let consumer = tokio::spawn(async move {
        while rx.changed().await.is_ok() {
            let value = *rx.borrow();
            println!("收到值: {}", value);
        }
    });
    
    tokio::join!(producer, consumer);
    
    Ok(())
}
